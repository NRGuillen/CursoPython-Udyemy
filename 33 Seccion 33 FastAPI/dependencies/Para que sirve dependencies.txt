La dependencias sirven para decirle a FastAPI donde buscar la logica y eviar meterlo junto a las rutas

from dependencies.message_dependencies import get_message_service

DEPENDENCIES
def get_message_service() -> MessageService:
    return MessageService()

ROUTERS
@router.get('/', response_model=List[Messages]) # response_model nos indica que va a retornar, fuertemente tipado
def read_mesages(service: MessageService = Depends(get_message_service)):
    """Depends(get_message_service) retorna MessageService() -|> service : MessageService = MessageService() -|> service.get_all()
    El depends sirve para que FastAPI ejecute una funcion y utilizar el return de esa funcion como parametro, en este
    caso en get_message_service

    Como get_message_service retorna un objeto de MessageService, service pasa a ser un objeto de MessageService y
    nos permite utilizar la logica de sus metodos sin mezclarl la logica dentro de esta ruta

    service: MessageService = Depends(get_message_service) --> equivale a -->  service = MessageService
    """
    return service.get_all()
